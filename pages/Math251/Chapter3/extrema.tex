\begin{thm}
	\textbf{Extreme Value Theorem.} A function $f(\vec{x})$ is continuous on a closed and bounded (shorter word: Compact) set $S$ in $\mathbb{R}^n$. Then f must attain a maximum and a minimum in S. 
\end{thm}
